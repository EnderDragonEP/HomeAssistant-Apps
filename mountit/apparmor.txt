#include <tunables/global>

profile addon_mountit flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/mdns>
  #include <abstractions/samba>
  #include <abstractions/smbpass>
  #include <abstractions/bash>

  file,
  signal (send) set=(kill,term,int,hup,cont),
  mount,
  umount,
  remount,

  # Samba file ownership/permission handling on mounted drives
  capability chown,
  capability dac_override,
  capability dac_read_search,
  capability fowner,
  capability fsetid,

  # Service management (s6, smbpasswd/PAM)
  capability kill,
  capability setgid,
  capability setuid,
  capability audit_write,

  # smbd: bind port 445, raise fd limits
  capability net_bind_service,
  capability sys_resource,
  capability sys_nice,

  # Drive handling: mount(2), modprobe, raw disk access (hdparm/hd-idle), udev
  capability sys_admin,
  capability sys_module,
  capability sys_rawio,
  capability net_admin,
  capability mknod,

  # Networks
  network udp,
  network tcp,
  network dgram,
  network stream,
  network inet,
  network inet6,
  network netlink raw,
  network unix dgram,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,

  # Files required
  /dev/** mrwkl,
  /tmp/** mrkwl,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Data access
  /data/** rw,

  # suppress ptrace denials
  ptrace (trace,read) peer=docker-default,

  # docker daemon confinement
  signal (receive) set=(kill,term) peer=/usr/bin/docker,

}
